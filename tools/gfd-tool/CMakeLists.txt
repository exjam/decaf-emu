project(gfd-tool)

include_directories(".")
include_directories("../../src/libdecaf/src")

file(GLOB_RECURSE SOURCE_FILES *.cpp)
file(GLOB_RECURSE HEADER_FILES *.h)

add_executable(gfd-tool ${SOURCE_FILES} ${HEADER_FILES})
set_target_properties(gfd-tool PROPERTIES FOLDER tools)

target_link_libraries(gfd-tool
    common
    libdecaf
    libgfd
    ${EXCMD_LIBRARIES})

if(DECAF_VULKAN)
    target_link_libraries(gfd-tool
        bin2c_vulkan_shaders
        ${VULKAN_LIBRARIES}
        ${SPIRV_LIBRARIES})
endif()

install(TARGETS gfd-tool RUNTIME DESTINATION "${CMAKE_INSTALL_PREFIX}")
